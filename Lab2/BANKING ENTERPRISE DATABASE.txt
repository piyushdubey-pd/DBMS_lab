CREATE TABLE branch(branch_name char(10), branch_city char(15), assets real, PRIMARY KEY(branch_name));
CREATE TABLE accounts(accno int, branch_name char(10), balance real, PRIMARY key(accno));
CREATE TABLE depositor(customer_name char(20), customer_street char(20), amount real, PRIMARY KEY(customer_name));
CREATE TABLE loan(loan_number int, branch_name char(10), amount real, PRIMARY KEY(loan_number),FOREIGN KEY(branch_name) REFERENCES branch(branch_name));
CREATE TABLE borrower(customer_name char(20), loan_number int, PRIMARY KEY(customer_name), FOREIGN KEY(loan_number) REFERENCES loan(loan_number));
